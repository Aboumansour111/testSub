import logging
import requests
from datetime import datetime
from telegram import Update
from telegram.ext import Application, CommandHandler, ContextTypes

# توکن بات
BOT_TOKEN = "YOUR_BOT_TOKEN_HERE"

LAT = 35.7218583
LON = 51.3346954

logging.basicConfig(level=logging.INFO)

# نام‌های فارسی دقیق (با اضافه شدن غروب خورشید)
PRAYER_NAMES = {
    "imsak": "امساک",
    "fajr": "اذان صبح",
    "sunrise": "طلوع آفتاب",
    "dhuhr": "اذان ظهر",
    "aasr": "اذان عصر",
    "sunset": "غروب خورشید",         # ← غروب آفتاب (قبل از اذان مغرب)
    "maghrib": "اذان مغرب",           # ← شروع نماز مغرب
    "maghrib_fadilah": "فضیلت مغرب", # ← پایان فضیلت نماز مغرب
    "midnight": "نیمه‌شب شرعی"
}

def get_prayer_times():
    today = datetime.now()
    url = "https://www.praytime-almahdyoon.com/api/public/prayertimes/day"
    params = {
        "day": today.day,
        "month": today.month,
        "year": today.year,
        "latitude": LAT,
        "longitude": LON,
        "timezone": 3.5,
        "languageCode": "en"
    }
    headers = {
        "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36",
        "Accept": "application/json, text/plain, */*",
        "Referer": "https://www.praytime-almahdyoon.com/",
        "Origin": "https://www.praytime-almahdyoon.com",
        "Sec-Fetch-Mode": "cors",
        "Sec-Fetch-Site": "same-origin"
    }

    try:
        session = requests.Session()
        session.get("https://www.praytime-almahdyoon.com/", headers=headers, timeout=10)
        r = session.get(url, params=params, headers=headers, timeout=15)
        if r.status_code == 200:
            data = r.json()
            if "imsak" in data:
                return data
    except Exception as e:
        logging.error(f"خطا: {e}")
    return None

async def start(update: Update, context: ContextTypes.DEFAULT_TYPE):
    await update.message.reply_chat_action("typing")
    await update.message.reply_text("در حال دریافت اوقات شرعی امروز تهران... ⏳")

    data = get_prayer_times()

    if not data:
        await update.message.reply_text("خطا در دریافت اطلاعات. چند دقیقه دیگه دوباره /start بزن.")
        return

    date = data.get("date", datetime.now().strftime("%d/%m/%Y"))

    msg = f"اوقات شرعی امروز تهران\n"
    msg += f"{date}\n\n"

    # ترتیب استاندارد (با غروب، مغرب و فضیلت)
    order = ["imsak", "fajr", "sunrise", "dhuhr", "aasr", "sunset", "maghrib", "maghrib_fadilah", "midnight"]
    for key in order:
        if key in data and data[key]:
            name = PRAYER_NAMES.get(key, key)
            msg += f"{name}: {data[key]}\n"

    await update.message.reply_text(msg.strip())

def main():
    app = Application.builder().token(BOT_TOKEN).build()
    app.add_handler(CommandHandler("start", start))
    print("بات اوقات شرعی تهران (با غروب، مغرب و فضیلت) فعال شد!")
    app.run_polling()

if __name__ == "__main__":
    main()
